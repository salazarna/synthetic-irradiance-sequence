\begin{algorithm}[h!]
  \SetAlgoLined
  \DontPrintSemicolon
    \KwInput{$H \in \mathbb{R}^{nm}$ contains the irradiance data for each sky condition set (matrix of $n$ days and $m$ timstamps).}
    \KwOutput{$S \in \mathbb{R}^m$ is the synthetic irradiance data for the $m$ timestamps.}

    \For{t in timestamps}{

    \If{t is first timestamp}{
     1. $S_t = N\sim(\mu = \text{mean}(\widehat{H}), \sigma = \text{std}(\widehat{H}))$; ~ \text{$\widehat{H}$ is a vector of $n$ days for the $m=t$ timestamp.}
     }

     2. $S \leftarrow S_t$

    \Else{
    3. $\widetilde{H}$; ~ \text{$\widetilde{H} \subset \widehat{H}$ is a matrix of $n$ days and $m=[t-1, t]$ timestamps.}

    4. $CI = S_{t-1} \pm Z \frac{\sigma}{\sqrt{N}}$

    5. $S_t = N\sim(\mu = \text{mean}(\widetilde{H} \pm CI), \sigma = \text{std}(\widetilde{H} \pm CI))$

    6. $S \leftarrow S_t$
    }
    }
    return $S$

   \caption{Stochastic Method}
  \label{alg:stochastic}
  \end{algorithm}

\begin{algorithm}[h!]
  \SetAlgoLined
  \DontPrintSemicolon
    \KwInput{$H \in \mathbb{R}^{nm}$ contains the irradiance data for each sky condition set (matrix of $n$ days and $m$ timstamps).}
    \KwOutput{$S \in \mathbb{R}^m$ is the synthetic irradiance data for the $m$ timestamps.}

    \For{t in timestamps}{

    \If{t is first timestamp}{
     1. $S_t = U\sim(\widehat{H}))$; ~ \text{$\widehat{H}$ is a vector of $n$ days for the $m=t$ timestamp.}

     2. $S \leftarrow S_t$
     }

    \Else{
    3. $\widetilde{H}$; ~ \text{$\widetilde{H} \subset \widehat{H}$ is a matrix of $n$ days and $m=[t-1, t]$ timestamps.}

    4. $CI = S_{t-1} \pm Z \frac{\sigma}{\sqrt{N}}$

    5. $S_t = U\sim(\widetilde{H} \pm CI))$

    6. $S \leftarrow S_t$
    }
    }
    return $S$

   \caption{Bootstrap Method}
  \label{alg:bootstrap}
  \end{algorithm}
